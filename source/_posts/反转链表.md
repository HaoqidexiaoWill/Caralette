```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def reverseList(self, head: ListNode) -> ListNode:
        pre,cur = None,head
        while(cur):
            # 当前结点非空时，当前结点指向前驱结点。pre,cur接着往后走。
            cur.next,pre,cur= pre,cur,cur.next
        return pre
```
