```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        hash = {}
        for idx,num in enumerate(nums):
            # 如果两个数的和等于target,返回他们的索引
            if target - num in hash:
                return [hash[target-num],idx]
            # 不满足的话，把当前遍历的数存到字典里
            else:
                hash[num] = idx
        return None
```
