```python

# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def reversePrint(self, head: ListNode) -> List[int]:
        # 辅助栈
        stack = []
        while(head):
            stack.append(head.val)
            head = head.next
        return stack[::-1]
```
